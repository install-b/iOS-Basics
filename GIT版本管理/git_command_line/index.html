  • Git初始化

      //初始化一个空的本地仓库
  git init

  // 初始化共享版本库
  git init --bare

  • Git配置

    • 用户名和邮箱是方便git追踪commit的用户

    • 某个git仓库没有配置用户名和邮箱，使用全局的

      //配置用户名
git config  user.name "your name"
//配置邮箱
git config  user.email "your name@example.com"

//配置全局的用户名（个人）
git config  --global user.name  "your name"
//配置全局的邮箱（个人）
git config  --global user.email "your email name @example.com"

//查看当前所有配置
git config -l

  • Git常用命令

      // 克隆共享仓库中的项目到本地
  git clone https://github.com/fzq2016/fzq2016.github.io

  // 检查文件状态（红色,绿色->说明已经被提交到暂缓区，可以直接提交）
  git status
  // 查看当前路径所有文件最新改动的地方：
  git diff
  // 查看版本信息的加强版本（增加了版本回退记录）
  git reflog
  // 查看版本信息的自定义版本（需配置全局自定义版本命令才能使用）
  git lg
  // 查看远程服务器地址和仓库名称
  git remote -v

  // 查看本地分支
  git branch
  // 查看远程分支：
  git branch -r
  // 创建本地分支：注意新分支创建后不会自动切换为当前分支
  git branch [name]
  // 切换分支
  git checkout [name]

  // 创建忽略文件
  touch .gitignore

  // 把忽略文件提交到暂缓区
  git add .gitignore
  // 表示将所有修改文件都提交到暂缓区
  git add .

  // 删除文件
  git rm 1                                                // 删除多个文件
  git rm 1 2 3

  // 提交修改到本地仓库
  git commit -m "添加了main.m文件" main.m
  // 文件可省略，省略则表示将所有修改文件提交暂缓区
  git commit -m "添加了main.m文件"

  // 切换到某个分支
  git checkout develop

  // 抓取远程仓库所有分支更新并合并到本地
  git pull

  // 将所有分支文件提交到远程仓库
  git push
  // 将本地主分支推到远程主分支
  git push origin master
  // 将本地主分支推到远程(如无远程主分支则创建，用于初始化远程仓库)
  git push -u origin master

  • Git命令起别名

      // 给status设置别名 简写为st
  git config alias.st "status"
  // 测试别名是否有效
  git st

  // 给commit -m指令起别名
  git config alias.ci "commit -m"

  // 设置全局别名（推荐将下面Git命令都改别名）
  git config --global alias.st "status
  git config --global alias.br "branch"
  git config --global alias.ci "commit -m"
  git config --global alias.co "checkout"

  git config --global alias.lg "log --color --graph --pretty=format:'%Cred%h%Creset -%C(yellow)%d%Creset %s %Cgreen(%cr) %C(bold blue)<%an>%Creset' --abbrev-commit"

  • Git查看版本信息

      // 查看git仓库的版本信息
  git log

  // 查看版本信息的加强版本（增加了版本回退记录）
  git reflog

  // 给log设置全局的别名，并设置颜色等
  git config --global alias.lg "log --color --graph --pretty=format:'%Cred%h%Creset -%C(yellow)%d%Creset %s %Cgreen(%cr) %C(bold blue)<%an>%Creset' --abbrev-commit"

  • Git版本操作

      // 回到当前版本，放弃所有没有提交的修改
  git reset --hard HEAD

  // 回到上一个版本
  git reset --hard HEAD^

  //回到上上个版本
  git reset --hard HEAD^^

  // 回到之前第n个修订版本（这里是回到前3个版本）
  git reset --hard HEAD~3

  //回到指定版本号的版本 (5位版本号)
  git reset --hard e695b67

  • Git远程仓库管理

      // 查看远程服务器地址和仓库名称
  git remote -v

  // 查看远程服务器仓库状态
  git remote show origin

  // 添加远程仓库地址
  git remote add origin git@ github:robbin/robbin_site.git

  // 设置远程仓库地址(用于修改远程仓库地址)
  git remote set-url origin git@ github.com:robbin/robbin_site.git

  // 删除远程仓库
  git remote rm <repository>

  • Git分支(branch)操作

      // 查看本地分支
  git branch

  // 查看远程分支：
  git branch -r

  // 创建本地分支：注意新分支创建后不会自动切换为当前分支
  git branch [name]

  // 切换分支
  git checkout [name]

  // 创建新分支并立即切换到新分支
  git checkout -b [name]

  // 删除分支
  // -d选项只能删除已经参与了合并的分支，对于未有合并的分支是无法删除的。如果想强制删除一个分支，可以使用-D选
  git branch -d [name]

  // 合并分支
  // 将名称为[name]的分支与当前分支合并
  git merge [name]

  // 创建远程分支(本地分支push到远程)
  git push origin [name]

  // 删除远程分支
  git push origin :heads/[name] 或 git push origin :[name]

  // 创建空的分支：(执行命令之前记得先提交你当前分支的修改，否则会被强制删干净没得后悔)
  git symbolic-ref HEAD refs/heads/[name]
  rm .git/index
  git clean -fdx

  • Git版本(tag)操作

      // 查看版本
  git tag

  // 创建版本
  git tag [name]

  // 删除版本
  git tag -d [name]

  // 查看远程版本
  git tag -l

  // 创建远程版本(本地版本push到远程)
  git push origin [name]

  // 删除远程版本
  git push origin :refs/tags/[name]

  // 合并远程仓库的tag到本地
  git pull origin --tags

  // 上传本地tag到远程仓库
  git push origin --tags

  // 创建带注释的tag
  git tag -a [name] -m 'yourMessage'
